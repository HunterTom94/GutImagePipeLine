167  def combine_pearson_AUC(a):
168      if a[0] <= 0 or a[1] <= 0:
169          out = 0
170      else:
171          out = a[0]*a[1]
172      return out

174  def product2label(y_df):
175      keep = y_df[(y_df['drv_pearson'] > 0) & (y_df['AUC'] > 0)]
176      throw = y_df[(y_df['drv_pearson'] <= 0) | (y_df['AUC'] <= 0)]
177      data = np.log(np.multiply(keep['AUC'].values, keep['drv_pearson'].values))
178  
179      mixture = GaussianMixture(n_components=3).fit(data.reshape(-1, 1))
180      means_hat = mixture.means_
181      sds_hat = np.sqrt(mixture.covariances_)
182  
183      g_ind = np.argsort(means_hat.flatten())[::-1][0]
184      c = np.exp(means_hat[g_ind] - 1.5 * sds_hat[g_ind][0])
185  
186      keep['label'] = keep['drv_pearson'] > c / keep['AUC']
187      throw['label'] = False
188      sti_df = pd.concat((keep, throw)).sort_index()
189  
190      return sti_df
